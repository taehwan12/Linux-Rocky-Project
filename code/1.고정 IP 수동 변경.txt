# 네트워크 인터페이스 장치 이름 확인 (예: ens160)
nmcli device

# 고정 IPv4 주소, 게이트웨이 및 전용 DNS 매핑 설정
nmcli connection modify ens160 ipv4.addresses 192.168.111.100/24 \
  ipv4.gateway 192.168.111.2 \
  ipv4.dns "192.168.111.100 8.8.8.8" \
  ipv4.method manual

# 네트워크 커넥션 재가동 및 적용 확인
nmcli connection up ens160
ifconfig ens160
