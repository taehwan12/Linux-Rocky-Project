# mysql_secure_installation

# 외부 원격 접속 권한 및 데이터베이스 생성 예시 (CLI 내부 진입 후)
mysql -u root -p -e "
CREATE DATABASE appdb;
CREATE USER 'admin'@'%' IDENTIFIED BY 'Password123!';
GRANT ALL PRIVILEGES ON appdb.* TO 'admin'@'%';
FLUSH PRIVILEGES;"
